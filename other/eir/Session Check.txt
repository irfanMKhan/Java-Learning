### SESSION-QUERY

 

#Total Session

 

select count(*) from SYN_SESSION_INFO

 

#Sigtran Session

 

select status, count(*) from SYN_SESSION_INFO where MACHINE IN ('PBVL-DEIR-SIGNALING-1', 'PBVL-DEIR-SIGNALING-2') GROUP BY STATUS

 

#Diameter Session

 

select status, count(*) from SYN_SESSION_INFO where MACHINE IN ('PBVL-DEIR-SIGNALING-3', 'PBVL-DEIR-SIGNALING-4') GROUP BY STATUS
